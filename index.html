library(tidyverse)

storms %>% 
  group_by(year) %>% 
  summarize(storms = unique(name)) %>% 
  summarize(canes = n()) %>% 
  ggplot(aes(year, canes))+
  geom_line(color = "darkred")+
  geom_point()+
  theme_classic()+
  labs(
    title = "Number of Hurricanes per Year",
    x = "Year",
    y = "Total Number of Storms",
    caption = "Source: storms dataset"
  )+
  theme(
    text = element_text(face = "bold")
  )
